Приведем еще один пример короткого алгоритма без переноса на другую страницу:

\begin{spacing}{0.8} % Поскольку стоит растяжение межстрочного интервала в тексте, то тут надо сжать
\begin{algorithm}
	\caption{\raggedright Euclid’s algorithm}\label{euclid}
	\begin{algorithmic}[1]
		\Procedure{Euclid}{$a,b$}\Comment{The g.c.d. of a and b}
		\State $r\gets a\bmod b$
		\While{$r\not=0$}\Comment{We have the answer if r is 0}
		\State $a\gets b$
		\State $b\gets r$
		\State $r\gets a\bmod b$
		\EndWhile\label{euclidendwhile}
		\State \Return $b$\Comment{The gcd is b}
		\EndProcedure
	\end{algorithmic}
\end{algorithm}
\end{spacing}
